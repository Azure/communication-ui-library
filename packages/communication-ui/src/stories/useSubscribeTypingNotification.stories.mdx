# useSubscribeTypingNotification

The `useSubscribeTypingNotification` hook allows components to subscribe to typing notifications. The hook accepts a
callback with the signature `(notification: TypingNotification) => void`. The callback is a required parameter. The
callback will be invoked when a typing notification occurs.

## Importing

```javascript
import { useSubscribeTypingNotification } from '@azure/communication-ui';
```

## Dependencies

- ChatProvider

## Usage

```jsx
import React from 'react';
import {
  ChatProvider,
  useSubscribeTypingNotification
} from '@azure/communication-ui';

const Example = () => (
  <ChatProvider config={token: <TOKEN>, displayName: <DISPLAYNAME>, endpointUrl: <URL>, threadId: <THREADID>}>
    <App />
  </ChatProvider>
);

const App = () => {
  useEffect(() => {
    useSubscribeTypingNotification((notification: TypingNotification) => {
      console.log(notification);
    });
  }, []);
}
```
